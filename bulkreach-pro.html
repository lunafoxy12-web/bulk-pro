🔧 Proxy Server Setup (if direct mode fails)
If your browser blocks the direct TextBelt call with a CORS error, deploy one of these tiny proxy scripts on any server (VPS, shared hosting, Vercel, Railway, etc.), then paste the URL above.
// server.js — Node.js proxy for TextBelt
// npm install express node-fetch cors
// node server.js

const express = require('express');
const fetch = require('node-fetch');
const cors = require('cors');
const app = express();
app.use(cors());
app.use(express.json());

app.post('/proxy/textbelt', async (req, res) => {
  try {
    const { phone, message, key } = req.body;
    const response = await fetch('https://textbelt.com/text', {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({ phone, message, key })
    });
    const data = await response.json();
    res.json(data);
  } catch (err) {
    res.status(500).json({ success: false, error: err.message });
  }
});

app.listen(3000, () => console.log('TextBelt proxy running on port 3000'));
// Paste this URL in BulkReach: http://YOUR_SERVER_IP:3000/proxy/textbelt

# app.py — Python Flask proxy for TextBelt
# pip install flask flask-cors requests
# python app.py

from flask import Flask, request, jsonify
from flask_cors import CORS
import requests

app = Flask(__name__)
CORS(app)

@app.route('/proxy/textbelt', methods=['POST'])
def proxy():
    data = request.get_json()
    try:
        r = requests.post('https://textbelt.com/text', data={
            'phone':   data.get('phone', ''),
            'message': data.get('message', ''),
            'key':     data.get('key', '')
        })
        return jsonify(r.json())
    except Exception as e:
        return jsonify({'success': False, 'error': str(e)}), 500

if __name__ == '__main__':
    app.run(port=3000)
